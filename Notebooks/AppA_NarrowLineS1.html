def critical_val(size1, size2, alpha=0.05):
    """Critical value for the KS two sample test"""
    section1 = -np.log(alpha/2)
    section2 = 1+(size2/size1)
    section3 = 2*size2
    return np.sqrt(section1*(section2/section3))
